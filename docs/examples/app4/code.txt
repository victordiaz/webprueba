// rgb selector
var r = 0
var g = 0
var b = 0

ui.addKnob(0.35, 0.15, 0.3, 0.3).range(0, 255).onChange(function (e) {
  r = e.value
  updateBackground()
})

ui.addKnob(0.35, 0.35, 0.3, 0.3).range(0, 255).onChange(function (e) {
  g = e.value
  updateBackground()
})

ui.addKnob(0.35, 0.55, 0.3, 0.3).range(0, 255).onChange(function (e) {
  b = e.value
  updateBackground()
})

function updateBackground() {
  ui.background(r, g, b)
}
